import { LinearGradient } from "expo-linear-gradient";
import { router, useLocalSearchParams } from "expo-router";
import * as Haptics from "expo-haptics";
import React, { useCallback, useMemo } from "react";
import { Pressable, StyleSheet, Text, View } from "react-native";
import { SafeAreaView } from "react-native-safe-area-context";

import { ChoiceButton } from "@/components/choice-button";
import { getNextSceneId, sceneConfigs, sceneOrder, type SceneId } from "@/constants/story";
import { useStory } from "@/providers/story-provider";

function isSceneId(value: string): value is SceneId {
  return sceneOrder.includes(value as SceneId);
}

export default function SceneScreen() {
  const params = useLocalSearchParams<{ scene?: string }>();
  const { filledCount, setChoice } = useStory();

  const sceneId = params.scene;

  const config = useMemo(() => {
    if (!sceneId || !isSceneId(sceneId)) {
      return null;
    }
    return sceneConfigs[sceneId];
  }, [sceneId]);

  const handleChoicePress = useCallback(
    (value: string) => {
      if (!config) return;

      console.log("[SceneScreen] choice selected", {
        scene: config.id, key: config.choiceKey, value,
      });

      setChoice(config.choiceKey, value);
      void Haptics.selectionAsync();

      const nextSceneId = getNextSceneId(config.id);
      if (nextSceneId) {
        router.push(`/${nextSceneId}`);
        return;
      }
      router.push("/recap");
    },
    [config, setChoice],
  );

  const handleBack = useCallback(() => {
    console.log("[SceneScreen] back pressed");
    router.back();
  }, []);

  if (!config) {
    console.log("[SceneScreen] invalid scene, redirecting", { sceneId });
    router.replace("/");
    return null;
  }

  const progress = (config.order / sceneOrder.length) * 100;

  return (
    <LinearGradient colors={[config.palette.background, "#FFFFFF"]} style={styles.container}>
      <SafeAreaView style={styles.safeArea}>
        <View style={styles.topRow}>
          <Pressable onPress={handleBack} style={styles.backButton}>
            <Text style={styles.backText}>Back</Text>
          </Pressable>
          <Text style={[styles.counter, { color: config.palette.muted }]}>
            {config.order}/{sceneOrder.length}
          </Text>
        </View>

        <View style={styles.content}>
          <View style={styles.progressWrap}>
            <View style={styles.progressTrack}>
              <View style={[styles.progressFill, { width: `${progress}%`, backgroundColor: config.palette.highlight }]} />
            </View>
            <Text style={[styles.progressLabel, { color: config.palette.muted }]}>{config.progressLabel}</Text>
          </View>

          <View style={[styles.card, { backgroundColor: config.palette.card }]}>
            <View style={styles.kickerRow}>
              <Text style={[styles.kicker, { color: config.palette.highlight }]}>{config.kicker}</Text>
              <View style={[styles.miniBadge, { backgroundColor: config.palette.background }]}>
                <Text style={[styles.miniBadgeText, { color: config.palette.text }]}>{filledCount} saved</Text>
              </View>
            </View>

            <Text style={[styles.title, { color: config.palette.text }]}>{config.title}</Text>
            <Text style={[styles.question, { color: config.palette.text }]}>{config.question}</Text>
            <Text style={[styles.caption, { color: config.palette.muted }]}>Pick the vibe and let the content arc continue.</Text>
          </View>

          <View style={styles.choiceStack}>
            {config.choices.map((choice) => (
              <ChoiceButton
                accent={choice.accent}
                description={choice.description}
                key={choice.id}
                label={choice.label}
                onPress={() => handleChoicePress(choice.label)}
              />
            ))}
          </View>
        </View>
      </SafeAreaView>
    </LinearGradient>
  );
}

  }, []);

  const resetChoices = useCallback(() => {
    console.log("[StoryProvider] resetChoices");
    setChoices(initialStoryChoices);
  }, []);

  const filledCount = useMemo(() => {
    return Object.values(choices).filter((value) => typeof value === "string" && value.length > 0).length;
  }, [choices]);

  return useMemo(() => ({
    choices, filledCount, setChoice, resetChoices,
  }), [choices, filledCount, resetChoices, setChoice]);
});

interface AppErrorBoundaryState { hasError: boolean; }

export class AppErrorBoundary extends React.Component<AppErrorBoundaryProps, AppErrorBoundaryState> {
  public state: AppErrorBoundaryState = { hasError: false };

  public static getDerivedStateFromError(): AppErrorBoundaryState {
    return { hasError: true };
  }

  public componentDidCatch(error: Error): void {
    console.log("[AppErrorBoundary] caught error", error);
  }

  private handleReload = (): void => {
    console.log("[AppErrorBoundary] retry requested");
    this.setState({ hasError: false });
  };

    if (this.state.hasError) {

utils/recap.ts

utility 
import { generateText } from "@rork-ai/toolkit-sdk";
import type { StoryChoices } from "@/constants/story";

function formatChoices(choices: StoryChoices): string {
  return [
    `Morning: ${choices.morning_choice ?? ""}`,
    `First Task: ${choices.task_choice ?? ""}`,
    `Midday: ${choices.midday_choice ?? ""}`,
    `Afternoon: ${choices.afternoon_choice ?? ""}`,
    `Evening: ${choices.evening_choice ?? ""}`,
  ].join("\n");
}

export function buildFallbackRecap(choices: StoryChoices): string {
  const morning = choices.morning_choice ?? "a mysterious morning";
  const task = choices.task_choice ?? "an equally mysterious task";
  const midday = choices.midday_choice ?? "some midday chaos";
  const afternoon = choices.afternoon_choice ?? "an afternoon plot twist";
  const evening = choices.evening_choice ?? "an iconic sign-off";

  return [
    `Okay, vlog recap: the day opened with ${morning.toLowerCase()}, which instantly set the tone...`,
    `From there, the first major plotline was ${task.toLowerCase()}, and the productivity arc had range.`,
    `By midday, ${midday.toLowerCase()} entered the chat and gave the storyline chaos in the best way.`,
    `The afternoon pivoted into ${afternoon.toLowerCase()}, which felt cinematic, efficient, or both.`,
    `Then the evening landed on ${evening.toLowerCase()}, exactly the kind of ending a creator day deserves.`,
    "Somehow the whole thing felt busy, funny, and weirdly well-edited in real time.",
    "Ten out of ten vlog energy, zero boring moments, and at least one sequence that deserved a slow zoom.",
  ].join(" ");
}

export async function generateRecap(choices: StoryChoices): Promise<string> {
  const prompt = [
    "Create a short day-in-the-life story based on these choices:",
    formatChoices(choices),
    "",
    "Write it in a fun, narrative style (6-8 sentences).",
    "Make it feel like a creator vlog recap.",
    "Keep it warm, engaging, and lightly humorous.",
    "Friendly. Creator-vlog energy. Short, punchy sentences. No moralizing or judgment.",
  ].join("\n");

  console.log("[generateRecap] generating recap", choices);

  try {
    const text = await generateText(prompt);
    const cleanedText = text.trim();
    console.log("[generateRecap] success", { length: cleanedText.length });
    return cleanedText.length > 0 ? cleanedText : buildFallbackRecap(choices);
  } catch (error) {
    console.log("[generateRecap] fallback triggered", error);
    return buildFallbackRecap(choices);
  }
}
